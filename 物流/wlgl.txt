{

    FILE *fp;

    int i;
    /*录入文本文档*/
    printf("输入文件名:");
    start=clock();
    gets(filename);
    fp=fopen(filename,"r");
    fseek(fp,72L, SEEK_SET);  /*从第二行开始读取*/
    for(i=0;i<100;i++)
    {
        fscanf(fp,"%s %s %s %f %s %f %f %f %f",c[i].type_name,c[i].cfcity_name,c[i].ddcity_name,

    scanf("%s",bgcity_name);

    start3=clock();
    printf("请输入包裹重量:\n");
    float weight;
    scanf("%f",&weight);
    printf("请输入包裹类型:\n");

                }
            }
        }
        else if(strcmp(type,"轻货")==0)

        else
            printf("输入错误！");
    }
